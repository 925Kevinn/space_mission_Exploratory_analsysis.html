import pandas as pd
import matplotlib.pyplot as plt

filename = 'D:/Course in Queensland/S2_1/DATA7901 Capstone Project/themed project/dataset_space_API_past_launches_raw.csv'
df = pd.read_csv(filename)

# add a column named Outcome_Value
# convert "Outcome" to 1 and 0 represent the true and false landing
df['Outcome_Value']=0
df1 = df.values
for i in range(len(df)):
    if "True" in df1[i,6]:
        df1[i,17] = 1 
dataset=pd.DataFrame(df1)

# Annual launch success rate
# split the date from year-month-day to year
year_occ = []
for i in range(len(df)):
    df1[i, 1] = df1[i, 1].split("-")[0]
dataset1=pd.DataFrame(df1)

# create two dictionary year_count_dist and success_count_dist to
# calculate the total number of launches per year
# and the number of successful launches per year
year_count_dist = dataset1[1].value_counts(sort = False).to_dict()
success_count_dist = {el:0 for el in year_count_dist.keys()}

for i in range(len(df)):
    for j in year_count_dist.keys():
        if df1[i,1] == j and df1[i,17] == 1:
            success_count_dist[j] += 1 


success_count = list(success_count_dist.values())
year_count = list(year_count_dist.values())
rate = [success_count[i]/year_count[i] for i in range(len(year_count))]

# plot the line chart
fig = plt.figure(dpi=128, figsize=(10,6))
plt.plot([i for i in year_count_dist.keys()],rate, marker = 'o')
plt.xlabel('year', fontsize=16)
plt.ylabel('Probability of successful launch', fontsize=16)
plt.title('Annual launch success rate', fontsize=24)
plt.show()

import numpy as np
from sklearn.model_selection import train_test_split
from sklearn import preprocessing
from sklearn.neighbors import KNeighborsClassifier


# Data imputation use mean value or adjacent value
#df['PayloadMass'] = df['PayloadMass'].fillna(df['PayloadMass'].mean())
df['PayloadMass'] = df['PayloadMass'].fillna(method ='bfill')

labels = np.array(df[["Orbit"]])
data = np.array(df[['PayloadMass']])


# Divide the dataset into train(0.7) and test(0.3)
train_data, test_data, train_labels, test_labels = train_test_split(data, labels, test_size = 0.3,random_state = 3)

# Data Normalization
norm_train = preprocessing.StandardScaler().fit(train_data)
train_data = norm_train.transform(train_data)
norm_test = preprocessing.StandardScaler().fit(test_data)
test_data = norm_test.transform(test_data)


# KNN
train_score = []
test_score = []
k = range(1,21,1)
for i in k:
    knn_clf = KNeighborsClassifier(n_neighbors = i)
    knn_clf.fit(train_data, train_labels.ravel())
    test_score.append(knn_clf.score(test_data, test_labels))
plt.xticks(k)
plt.plot(k, test_score, 'o-', label = 'Test data')
plt.legend()
plt.xlabel('Number of the neighbor')
plt.ylabel('Classification Correct rate')
plt.title('KNN classification with different neighbors')
plt.show()
